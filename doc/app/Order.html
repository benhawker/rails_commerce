def self.new_from_cart(cart)
  order = Order.new
  cart.cart_items.each { |x|
    order.order_items.append(OrderItem.create(
                               product_id: x.product_id,
                               quantity: x.quantity,
                               price: x.product.price
    ))
  }
  return order
end
